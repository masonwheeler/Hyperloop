[docs]def random_subset_track(a, N, K):
    used = [False] * N
    result = []

    for i in range(0, K):
        # get a random element between 0 and N.
        j = int(random.random() * N)
        while (used[j]):
            j = int(random.random() * N)
        result.append(j)
        used[j] = True

    return result
